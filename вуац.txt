-- –§—É–Ω–∫—Ü–∏—è –ø–æ–∏—Å–∫–∞ —Ç–∞–±–ª–∏—á–∫–∏
local function findPlayerSign(username)
    -- –ò—â–µ–º —Ç–∞–±–ª–∏—á–∫—É —Å —Ä–∞–∑–Ω—ã–º–∏ –≤–∞—Ä–∏–∞–Ω—Ç–∞–º–∏ –∏–º–µ–Ω
    local patterns = {
        username .. "'s Sign",
        username .. "'s sign", 
        username .. " Sign",
        username .. " sign",
        username .. "_Sign"
    }
    
    for _, pattern in ipairs(patterns) do
        local sign = workspace:FindFirstChild(pattern, true)
        if sign and sign:IsA("Model") then
            return sign
        end
    end
    
    return nil
end

-- –§—É–Ω–∫—Ü–∏—è —É–¥–∞–ª–µ–Ω–∏—è —Ç–∞–±–ª–∏—á–∫–∏
local function deleteSign(username)
    -- –ü—Ä–æ–≤–µ—Ä—è–µ–º, –Ω–µ —É–¥–∞–ª–µ–Ω–∞ –ª–∏ —É–∂–µ —Ç–∞–±–ª–∏—á–∫–∞
    if deletedSigns[username] then
        -- –ï—Å–ª–∏ —Ç–∞–±–ª–∏—á–∫–∞ —É–∂–µ —É–¥–∞–ª–µ–Ω–∞, –Ω–∞—Ö–æ–¥–∏–º –µ—ë —Å–Ω–æ–≤–∞
        local sign = findPlayerSign(username)
        if not sign then
            return false, "–¢–∞–±–ª–∏—á–∫–∞ —É–∂–µ —É–¥–∞–ª–µ–Ω–∞ –∏ –Ω–µ –Ω–∞–π–¥–µ–Ω–∞ –≤ workspace"
        end
        
        -- –û–±–Ω–æ–≤–ª—è–µ–º –¥–∞–Ω–Ω—ã–µ –æ–± —É–¥–∞–ª–µ–Ω–Ω–æ–π —Ç–∞–±–ª–∏—á–∫–µ
        deletedSigns[username] = {
            Clone = sign:Clone(),
            OriginalParent = sign.Parent,
            Position = sign:GetPivot().Position,
            Time = os.time()
        }
        
        -- –î–µ–ª–∞–µ–º —Ç–∞–±–ª–∏—á–∫—É –Ω–µ–≤–∏–¥–∏–º–æ–π
        for _, part in ipairs(sign:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Transparency = 1
                part.CanCollide = false
            elseif part:IsA("Decal") or part:IsA("Texture") then
                part.Transparency = 1
            end
        end
        
        return true, "–¢–∞–±–ª–∏—á–∫–∞ –ø–æ–≤—Ç–æ—Ä–Ω–æ —É–¥–∞–ª–µ–Ω–∞: " .. username
    end
    
    local sign = findPlayerSign(username)
    if not sign then
        return false, "–¢–∞–±–ª–∏—á–∫–∞ –Ω–µ –Ω–∞–π–¥–µ–Ω–∞"
    end
    
    -- –°–æ—Ö—Ä–∞–Ω—è–µ–º –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –¥–ª—è –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏—è
    deletedSigns[username] = {
        Clone = sign:Clone(),
        OriginalParent = sign.Parent,
        Position = sign:GetPivot().Position,
        Time = os.time()
    }
    
    -- –î–µ–ª–∞–µ–º —Ç–∞–±–ª–∏—á–∫—É –Ω–µ–≤–∏–¥–∏–º–æ–π
    for _, part in ipairs(sign:GetDescendants()) do
        if part:IsA("BasePart") then
            part.Transparency = 1
            part.CanCollide = false
        elseif part:IsA("Decal") or part:IsA("Texture") then
            part.Transparency = 1
        end
    end
    
    return true, "–¢–∞–±–ª–∏—á–∫–∞ —É–¥–∞–ª–µ–Ω–∞: " .. username
end

-- –§—É–Ω–∫—Ü–∏—è –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏—è —Ç–∞–±–ª–∏—á–∫–∏
local function restoreSign(username)
    if not deletedSigns[username] then
        return false, "–¢–∞–±–ª–∏—á–∫–∞ –Ω–µ –±—ã–ª–∞ —É–¥–∞–ª–µ–Ω–∞"
    end
    
    local signData = deletedSigns[username]
    
    -- –ù–∞—Ö–æ–¥–∏–º —Ç–µ–∫—É—â—É—é —Ç–∞–±–ª–∏—á–∫—É –≤ workspace
    local currentSign = findPlayerSign(username)
    if currentSign then
        -- –í–æ—Å—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –≤–∏–¥–∏–º–æ—Å—Ç—å —Å—É—â–µ—Å—Ç–≤—É—é—â–µ–π —Ç–∞–±–ª–∏—á–∫–∏
        for _, part in ipairs(currentSign:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Transparency = 0
                part.CanCollide = true
            elseif part:IsA("Decal") or part:IsA("Texture") then
                part.Transparency = 0
            end
        end
        
        -- –û–±–Ω–æ–≤–ª—è–µ–º –∫–ª–æ–Ω –¥–ª—è –±—É–¥—É—â–∏—Ö —É–¥–∞–ª–µ–Ω–∏–π
        signData.Clone = currentSign:Clone()
        signData.Time = os.time()
        
        return true, "–¢–∞–±–ª–∏—á–∫–∞ –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∞: " .. username
    else
        -- –ï—Å–ª–∏ —Ç–∞–±–ª–∏—á–∫–∏ –Ω–µ—Ç –≤ workspace, —Å–æ–∑–¥–∞–µ–º –∏–∑ –∫–ª–æ–Ω–∞
        local newSign = signData.Clone:Clone()
        newSign.Parent = signData.OriginalParent
        
        -- –û–±–Ω–æ–≤–ª—è–µ–º –∫–ª–æ–Ω
        signData.Clone = newSign:Clone()
        signData.Time = os.time()
        
        return true, "–¢–∞–±–ª–∏—á–∫–∞ –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∞ –∏–∑ –∫–ª–æ–Ω–∞: " .. username
    end
end

-- –§—É–Ω–∫—Ü–∏—è –≤–∫–ª—é—á–µ–Ω–∏—è/–≤—ã–∫–ª—é—á–µ–Ω–∏—è ESP
local function toggleESP()
    espEnabled = not espEnabled
    
    -- –û—á–∏—â–∞–µ–º –ø—Ä–µ–¥—ã–¥—É—â–∏–µ ESP –æ–±—ä–µ–∫—Ç—ã
    for _, obj in pairs(espObjects) do
        if obj and obj.Parent then
            obj:Destroy()
        end
    end
    espObjects = {}
    
    if espEnabled then
        statusLabel.Text = "ESP –≤–∫–ª—é—á–µ–Ω"
        statusIcon.Text = "üîÜ"
        highlightButton.Text = "üîÜ  –í–´–ö–õ. –ü–û–î–°–í–ï–¢–ö–£"
        highlightButton.BackgroundColor3 = Color3.fromRGB(200, 60, 60)
        
        -- –ü–æ–¥—Å–≤–µ—á–∏–≤–∞–µ–º –≤—Å–µ —Ç–∞–±–ª–∏—á–∫–∏ –≤ –∏–≥—Ä–µ
        for username, data in pairs(deletedSigns) do
            -- –î–ª—è —É–¥–∞–ª–µ–Ω–Ω—ã—Ö —Ç–∞–±–ª–∏—á–µ–∫
            local sign = findPlayerSign(username)
            if sign then
                local highlight = Instance.new("Highlight")
                highlight.Name = username .. "_DeletedESP"
                highlight.FillColor = Color3.fromRGB(255, 50, 50)
                highlight.OutlineColor = Color3.fromRGB(255, 100, 100)
                highlight.FillTransparency = 0.7
                highlight.OutlineTransparency = 0.3
                highlight.Adornee = sign.PrimaryPart or sign:FindFirstChildWhichIsA("BasePart")
                highlight.Parent = CoreGui
                table.insert(espObjects, highlight)
            end
        end
        
        -- –î–ª—è –æ–±—ã—á–Ω—ã—Ö —Ç–∞–±–ª–∏—á–µ–∫
        local players = Players:GetPlayers()
        for _, player in ipairs(players) do
            if not deletedSigns[player.Name] then
                local sign = findPlayerSign(player.Name)
                if sign then
                    local highlight = Instance.new("Highlight")
                    highlight.Name = player.Name .. "_ESP"
                    highlight.FillColor = Color3.fromRGB(0, 255, 0)
                    highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
                    highlight.FillTransparency = 0.8
                    highlight.OutlineTransparency = 0.2
                    highlight.Adornee = sign.PrimaryPart or sign:FindFirstChildWhichIsA("BasePart")
                    highlight.Parent = CoreGui
                    table.insert(espObjects, highlight)
                end
            end
        end
    else
        statusLabel.Text = "ESP –≤—ã–∫–ª—é—á–µ–Ω"
        statusIcon.Text = "‚úÖ"
        highlightButton.Text = "üîÜ  –ü–û–î–°–í–ï–¢–ò–¢–¨ –¢–ê–ë–õ–ò–ß–ö–ò"
        highlightButton.BackgroundColor3 = Color3.fromRGB(60, 140, 200)
    end
end

-- –§—É–Ω–∫—Ü–∏—è –∞–Ω–∏–º–∞—Ü–∏–∏ –∫–Ω–æ–ø–æ–∫
local function animateButton(button)
    local originalSize = button.Size
    local tweenInfo = TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
    
    local tween1 = TweenService:Create(button, tweenInfo, {Size = originalSize - UDim2.new(0, 5, 0, 5)})
    local tween2 = TweenService:Create(button, tweenInfo, {Size = originalSize})
    
    tween1:Play()
    tween1.Completed:Connect(function()
        tween2:Play()
    end)
end

-- –§—É–Ω–∫—Ü–∏—è –ø–µ—Ä–µ—Ç–∞—Å–∫–∏–≤–∞–Ω–∏—è –æ–∫–Ω–∞ (–ò–°–ü–†–ê–í–õ–ï–ù–ù–ê–Ø –í–ï–†–°–ò–Ø)
header.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = mainContainer.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

header.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement and dragging then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        mainContainer.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == Enum.KeyCode.G then
        local target = mouse.Target
        if target then
            local model = target:FindFirstAncestorOfClass("Model")
            if model then
                -- –ü—Ä–æ–≤–µ—Ä—è–µ–º, –Ω–µ —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —ç—Ç–æ —É–¥–∞–ª–µ–Ω–Ω–æ–π —Ç–∞–±–ª–∏—á–∫–æ–π
                for username, data in pairs(deletedSigns) do
                    if data.Clone.Name == model.Name then
                        ownerLabel.Text = "–í–ª–∞–¥–µ–ª–µ—Ü —Ç–∞–±–ª–∏—á–∫–∏: " .. username
                        searchBox.Text = username
                        statusLabel.Text = "–û–ø—Ä–µ–¥–µ–ª–µ–Ω–∞ —É–¥–∞–ª–µ–Ω–Ω–∞—è —Ç–∞–±–ª–∏—á–∫–∞: " .. username
                        statusIcon.Text = "‚úÖ"
                        return
                    end
                end
                
                -- –ü—Ä–æ–≤–µ—Ä—è–µ–º –æ–±—ã—á–Ω—ã–µ —Ç–∞–±–ª–∏—á–∫–∏
                if model.Name:find("'s Sign") or model.Name:find(" Sign") or model.Name:find("_Sign") then
                    local owner = model.Name:gsub("'s Sign", ""):gsub(" Sign", ""):gsub("_Sign", ""):gsub("'s sign", "")
                    ownerLabel.Text = "–í–ª–∞–¥–µ–ª–µ—Ü —Ç–∞–±–ª–∏—á–∫–∏: " .. owner
                    searchBox.Text = owner
                    statusLabel.Text = "–í–ª–∞–¥–µ–ª–µ—Ü –æ–ø—Ä–µ–¥–µ–ª–µ–Ω: " .. owner
                    statusIcon.Text = "‚úÖ"
                end
            end
        end
    end
end)

-- –í–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ –ø–æ—Å–ª–µ–¥–Ω–µ–π —É–¥–∞–ª–µ–Ω–Ω–æ–π —Ç–∞–±–ª–∏—á–∫–∏ –ø–æ Ctrl+Z
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    
    if input.KeyCode == Enum.KeyCode.Z and UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then
        local lastUsername, lastTime = nil, 0
        for username, data in pairs(deletedSigns) do
            if data.Time and data.Time > lastTime then
                lastUsername = username
                lastTime = data.Time
            end
        end
        
        if lastUsername then
            searchBox.Text = lastUsername
            ownerLabel.Text = "–í–ª–∞–¥–µ–ª–µ—Ü —Ç–∞–±–ª–∏—á–∫–∏: " .. lastUsername
            
            local success, message = restoreSign(lastUsername)
            statusLabel.Text = message
            statusIcon.Text = success and "‚úÖ" or "‚ùå"
            
            -- –û–±–Ω–æ–≤–ª—è–µ–º ESP –µ—Å–ª–∏ –æ–Ω –≤–∫–ª—é—á–µ–Ω
            if espEnabled then
                toggleESP() -- –í—ã–∫–ª—é—á–∞–µ–º
                toggleESP() -- –í–∫–ª—é—á–∞–µ–º —Å–Ω–æ–≤–∞
            end
        end
    end
end)
